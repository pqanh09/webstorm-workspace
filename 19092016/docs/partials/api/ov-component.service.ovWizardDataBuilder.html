Example

Example: 


 ctrl.wizard = ovWizardDataBuilder.getBuilder()
 .setSequencePages([step1, step2, step3])
 .setIdKey('test')
 .setCommitPage(stepFinish)//optional
 //.setCurrentPage(step2CareerRegistration) //optional, default: first page

 .setTitle('Registration')
 //.setTitleIconClasses('fa fa-info-circle') //optional
 //.setPageDescIconClasses('fa fa-info') //optional

 //.setShowCommit(true) //optional,  default: true
 //.setShowFinish(false) //optional,  default: false
 //.setShowHelp(true) //optional,  default: false
 //.setShowCancel(true) //optional,  default: true
 //.setHideClear(true)//optional,  default: true
 //.setOverrideClass('ov-adv-wz-demo-max-width-800') //optional

 //.setCommitLabel($i18next('button-label.create')) //optional,  default: 'Commit'
 //.setFinishLabel($i18next('button-label.ok'))//optional,  default: 'Finish'

 //.setCanCancel(canCancel) //optional
 //.setCanCommit(canCommit)//optional
 //.setCanFinish(canFinish)//optional

 .setPerformCommit(commit)//optional
 .setPerformFinish(finish)//optional
 //.setPerformNext(next)//optional
 .setPerformCancel(cancel)//optional
 //.setPerformHelp(help)//optional

 .build();
