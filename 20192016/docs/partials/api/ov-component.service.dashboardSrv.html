dashboardSrv
service in module ov-component

Description

Support adding more widgets

Methods

  • addWidgets(widgets)

    • add more widgets to dashboard
    • this function must be call in run function
    Parameters
    ParamTypeDetails
    widgetsobject

    contain all widgets object

    • widget object
      • appId {string} - used to check permission
      • title {string} - title of the widget
      • content {string} - path of the widget template
      • description {string} - description of the widget
      • imgUrl {string} - path of the sample image
      • ctrl {string} - widget controller name
      • refresh {boolean} - show/hide refresh button
      • close {boolean} - show/hide close button
      • movable {boolean} - enable/disable movable functionality

Example

Example: 


 (function () {
  'use strict';
  angular.module('wma.widget')
    .run(runFunction);

  runFunction.$inject = ['dashboardSrv', 'wmaWidgetConstant'];

  function runFunction(dashboardSrv, wmaWidgetConstant) {
    //add widgets
    var widgets = {
        wmaWidgetSample: {
          //appId: 'wma',//use to check permission
          title: 'WMA',
          content: 'wma/wma-apps/wma-widget/wma-widget-sample/wma-widget-sample.html',
          description: 'WMA description',
          imgUrl: 'path/sample.png',
          ctrl: 'WMAWidgetSampleController as ctrl',
          refresh: true,
          close: true,
          movable: true
        }
      }
    dashboardSrv.addWidgets(widgets);
  }
})();
