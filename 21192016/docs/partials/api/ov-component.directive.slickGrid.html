slickGrid
directive in module ov-component

Description

The component help to show data as tabular mode

Usage

as attribute
<ANY slick-grid
     grid-view-id="{string}"
     grid-columns="{array}">
   ...
</ANY>

Parameters

ParamTypeDetails
gridViewIdstring

component id

gridColumnsarray

array column

Example

//HTML
<div slick-grid
         grid-view-id="vm.gridViewId"
         grid-columns="vm.gridColumns">
</div>

//In JS
vm.gridViewId='component-id';


vm.data = [
   {
     key1: 'value1',
     key2: 'value2'
   }
];

var attrs = {
   column1: {
     key: 'key1',
     i18nKey: 'i18nKey1'
   },
   column2: {
     key: 'key2',
     i18nKey: 'i18nKey2'
   }
};
vm.gridColumns = ovDataViewService.getColumnList(attrs);

$scope.$on('onGridInit', function (event, gridElm, grid, dataView) {
   if (gridElm[0].id === vm.gridViewId) {
       vm.grid = grid;
       vm.dataView = dataView;

       vm.dataView.setItems(vm.data, 'instanceid');
   }
});
