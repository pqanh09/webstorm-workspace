Example

Basic example: 


   function showDialog() {
        var dialog = dlgDataBuilder.getBuilder()
          .setTitle('title')
          .setIdDlg('dialogId')
          .setTemplateUrl('your-url/dialog.html')
          .build();
        $ovDlgService2.showDialog($scope, dialog);
    };
