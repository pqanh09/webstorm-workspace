ovNestedList
directive in module ov-component

Description

Sidebar menu component

Usage

as element:
<ov-nested-list
       root="{object}"
       [selected]
       [on-selected-callback="{function}"]
       [on-pre-selected-callback="{function}"]
       [id-app="{string}"]>
</ov-nested-list>

Parameters

ParamTypeDetails
rootobject

data object.

selected
(optional)
object

current selected object.

onSelectedCallback
(optional)
function

call back function to be executed when select an item

onPreSelectedCallback
(optional)
function

call back function to be executed before when pre-select an item

idApp
(optional)
string

An ID to be added to the element to support automation test e.g. id-app="test"

Example

Basic Example: 

<!-- HTML -->
 <ov-nested-list
   id-app="test-id"
   root="ctrl.sideBarMenu">
 </ov-nested-list>

//JS
ctrl.sideBarMenu  = getSideBarMenu();

//get sidebar menu data
function getSideBarMenu(){
      var sideBarMenu = {
          id: 'demo',
          title: 'Demo',
          menuItems: []
        },
        ovUxSelect = {
          id: 'ov-ux-select',
          title: 'ovUxSelect',
          templateUrl: 'ov-component-demo/ov-ux-select-demo/ov-ux-select-demo.html',
          parent: sideBarMenu
        },
        ovDataView = {
          id: 'ov-data-view',
          title: 'ovDataViewDemo',
          templateUrl: 'ov-component-demo/ov-data-view/ov-data-view-demo.html',
          parent: sideBarMenu
        };

      sideBarMenu.menuItems= [ovUxSelect, ovDataView];

      return sideBarMenu;
    }
